## 一、關於「直接實作 BullMQ」的見解

### 您的建議
> 雖然文件建議 MVP 先做同步，但我建議直接從 BullMQ 開始實作。因為旅遊行程生成涉及多個外部 API（Scraping, LLM, Image Gen），網絡不穩定的機率很高，同步模式在測試階段就容易因為 Timeout 而讓人感到挫折。

### 我的見解：完全同意 ✅

**原因分析：**

1. **技術債務問題**
   - 如果先做同步版本，後續重構為非同步會產生大量技術債務
   - 需要重寫大部分代碼（前端輪詢、後端任務管理、錯誤處理）
   - 浪費開發時間

2. **測試階段的挫折感**
   - 圖片生成 API 可能因為負載而變慢（15-30 秒 → 60 秒）
   - LLM API 可能因為網絡問題超時
   - 同步模式下，前端會直接報錯，用戶體驗極差

3. **生產環境的穩定性**
   - 旅遊網站的行程生成是核心功能，不能因為 API 超時而崩潰
   - BullMQ 提供的「失敗重試」機制是必需品，不是奢侈品

4. **實作複雜度**
   - BullMQ 的學習曲線不高（1-2 小時即可上手）
   - 相比重構的成本，直接實作更划算

### 調整後的實作方案

#### 方案：直接實作 BullMQ + 非同步處理

**技術棧：**
- **任務隊列**：BullMQ（基於 Redis）
- **前端輪詢**：tRPC Subscription 或 WebSocket（即時更新）
- **進度追蹤**：Redis + Job Progress API

```
完成後通知前端
```

```typescript
// server/queue/tourGenerationQueue.ts
import { Queue, Worker, Job } from 'bullmq';
import Redis from 'ioredis';

// Redis 連接
const redis = new Redis({
  host: process.env.REDIS_HOST || 'localhost',
  port: parseInt(process.env.REDIS_PORT || '6379'),
  maxRetriesPerRequest: null // BullMQ 需要
});

// 任務隊列
export const tourGenerationQueue = new Queue('tour-generation', {
  connection: redis,
  defaultJobOptions: {
    attempts: 3, // 失敗後重試 3 次
    backoff: {
      type: 'exponential',
      delay: 2000 // 2 秒、4 秒、8 秒
    },
    removeOnComplete: 100, // 保留最近 100 個完成的任務
    removeOnFail: 500 // 保留最近 500 個失敗的任務
  }
});

// Worker
export const tourGenerationWorker = new Worker(
  'tour-generation',
  async (job: Job) => {
    const { url, userId } = job.data;
    
    try {
      // 階段 1：抓取網頁（10%）
      await job.updateProgress(10);
      await job.log('開始抓取網頁...');
      const scrapedData = await webScraperAgent.scrape({ url });
      
      // 階段 2：分析內容（30%）
      await job.updateProgress(30);
      await job.log('開始分析內容...');
      const tourData = await contentAnalyzerAgent.analyze(scrapedData);
      
      // 階段 3：生成配色和 Prompt（50%）
      await job.updateProgress(50);
      await job.log('生成配色和圖片 Prompt...');
      const [colorTheme, imagePrompts] = await Promise.all([
        colorThemeAgent.generate(tourData),
        imagePromptAgent.generateAll(tourData)
      ]);
      
      // 階段 4：生成圖片（70%）
      await job.updateProgress(70);
      await job.log('生成 Hero 圖片...');
      const heroImage = await imageGenerationAgent.generate({
        prompt: imagePrompts.hero,
        fileKey: `tours/${job.id}/hero.jpg`
      });
      
      // 階段 5：儲存到資料庫（90%）
      await job.updateProgress(90);
      await job.log('儲存到資料庫...');
      const [tour] = await db.insert(tours).values({
        ...tourData,
        heroImage,
        colorTheme: JSON.stringify(colorTheme),
        status: 'pending_review', // 待審核
        autoGenerated: true
      }).returning();
      
      // 完成（100%）
      await job.updateProgress(100);
      await job.log('完成！');
      
      return { tourId: tour.id };
      
    } catch (error) {
      await job.log(`錯誤：${error.message}`);
      throw error; // 讓 BullMQ 處理重試
    }
  },
  {
    connection: redis,
    concurrency: 3 // 最多同時處理 3 個任務
  }
);

// 監聽事件
tourGenerationWorker.on('completed', (job) => {
  console.log(`[任務 ${job.id}] 完成`);
});

tourGenerationWorker.on('failed', (job, error) => {
  console.error(`[任務 ${job?.id}] 失敗:`, error);
});
```

```typescript
export async function searchUnsplashImage(
  query: string,
```

```typescript
        schema: {
          type: "object",
          properties: {
            hero: { type: "string" },

            highlight2: { type: "string" },
            highlight3: { type: "string" }
          },
          required: ["hero", "highlight1", "highlight2", "highlight3"]
        }
      }
    }

  const prompts = JSON.parse(response.choices[0].message.content);
  
  // 驗證一致性
  validateStyleConsistency(prompts, styleGuide);
  
  return prompts;
}

// 驗證風格一致性
function validateStyleConsistency(
  prompts: ImagePrompts,
  styleGuide: StyleGuide
): void {
  const allPrompts = Object.values(prompts);
  
  // 檢查每個 Prompt 是否包含風格關鍵字
  for (const prompt of allPrompts) {
    const hasStyleKeyword = styleGuide.keywords.some(keyword =>
      prompt.toLowerCase().includes(keyword.toLowerCase())
    );
    
    if (!hasStyleKeyword) {
      throw new Error(`Prompt 缺少風格關鍵字：${prompt}`);
    }
  }
  
  // 檢查是否包含品質關鍵字
  const qualityKeywords = ['8K', '4K', 'ultra HD', 'professional'];
  for (const prompt of allPrompts) {
    const hasQuality = qualityKeywords.some(keyword =>
      prompt.includes(keyword)
    );
    
    if (!hasQuality) {
      throw new Error(`Prompt 缺少品質關鍵字：${prompt}`);
    }
  }
}
```

```typescript
async function verifyOriginality(
  generatedContent: string,
  originalContent: string
): Promise<{ isOriginal: boolean; similarity: number }> {
  // 計算相似度（使用 Levenshtein Distance 或其他演算法）
  const similarity = calculateSimilarity(generatedContent, originalContent);
  
  // 如果相似度超過 70%，視為抄襲
  if (similarity > 0.7) {
    return { isOriginal: false, similarity };
  }
  
  return { isOriginal: true, similarity };
}

async function analyze(data: any): Promise<TourData> {

  // 驗證原創性

    tourData.title,
    data.rawContent
  );
  
  if (!titleCheck.isOriginal) {
    console.warn(`[警告] 標題相似度過高（${titleCheck.similarity * 100}%），重新生成`);
    // 重新生成標題
    tourData.title = await regenerateTitle(data);

}
```
